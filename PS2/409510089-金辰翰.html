def port(returns, stds, rho):
    ri = np.array(returns)
    si = np.array(stds)
    rp = []
    sp = []
    for i in range(1000):
        w1 = (i-1)/1000
        w2 = (1-w1)*np.random.random()
        w3 = 1-(w1+w2)
        wtotal = np.array([w1, w2, w3])
        rp.append(sum(wtotal*ri))
        sp.append(np.sqrt(sum((si*wtotal)**2) + 2*rho[0]*wtotal[0]*wtotal[1]*si[0]*si[1] + 2*rho[1]*wtotal[1]*wtotal[2]*si[1]*si[2] + 2*rho[2]*wtotal[2]*wtotal[0]*si[2]*si[0]))
        
    pt = pd.DataFrame({'Return':rp, 'Risk': sp}) 
    return pt.plot(x = 'Risk', y = 'Return', kind = 'scatter')
